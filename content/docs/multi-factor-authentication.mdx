## Usage

```tsx
import { MultiFactor } from "@/components/ui/multi-factor-authentication";

export default function Page() {
  return <MultiFactor />;
}
```

### OTP Digits

Customize the digits that appear in the animation:

```tsx
const OTP_DIGITS = ["3", "5", "8", "1", "2", "9"]; // Change to any 6 digits
```
