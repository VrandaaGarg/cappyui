## Usage

```tsx
import { BookAppointment } from "@/components/components/book-appointment";

export default function Page() {
  // Define booked dates
  const bookedDates = [
    new Date(2025, 9, 15), // October 15, 2025
    new Date(2025, 9, 20), // October 20, 2025
    new Date(2025, 9, 25), // October 25, 2025
  ];

  return (
    <div className="min-h-screen bg-neutral-100 p-10">
      <BookAppointment bookedDates={bookedDates} />
    </div>
  );
}
```

## Props

| Prop          | Type     | Default | Description                                                                       |
| ------------- | -------- | ------- | --------------------------------------------------------------------------------- |
| `className`   | `string` | -       | Optional additional CSS classes                                                   |
| `bookedDates` | `Date[]` | `[]`    | Array of dates that are already booked (shown with strikethrough, non-selectable) |
