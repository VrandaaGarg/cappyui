## Usage

```tsx
import { ImagePuzzle } from "@/components/ui/image-puzzle";

export default function MyComponent() {
  return <ImagePuzzle />;
}
```

With custom image URL:

```tsx
import { ImagePuzzle } from "@/components/ui/image-puzzle";

export default function MyComponent() {
  return <ImagePuzzle imageSrc="https://example.com/your-image.jpg" />;
}
```

## Props

| Prop        | Type     | Default       | Description                               |
| ----------- | -------- | ------------- | ----------------------------------------- |
| `className` | `string` | -             | Optional additional CSS classes           |
| `imageSrc`  | `string` | Default image | URL of the image to display in the puzzle |
